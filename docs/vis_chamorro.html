import pandas as pd
import matplotlib.pyplot as plt

# 1. Load Data
df = pd.read_csv('/content/Peliculas Ghibli - Peliculas Ghibli.csv')

# 2. Process Data
# Create the category column
df['Category'] = df.apply(lambda row: 'Femenino' if row['Genero'] == 'Femenino'
                          else f"Masculino - {row['Protagonismo']}", axis=1)

# Calculate percentages and sort by Year ('Ano')
# We index by Year AND Movie to sort chronologically, then drop Year for plotting
plot_data = pd.crosstab(
    index=[df['Ano'], df['Pelicula']],
    columns=df['Category'],
    normalize='index'
) * 100

plot_data.index = plot_data.index.droplevel(0) # Keep only movie titles for x-axis

# 3. Visualization
custom_colors = {
    'Femenino': '#F4C6B8',
    'Masculino - Incidental': '#9FC6B8',
    'Masculino - Protagonista': '#BFD7EA',
    'Masculino - Secundario': '#D7C9E3'
}
# Map colors to the columns present in the data
plot_colors = [custom_colors.get(col, '#CCCCCC') for col in plot_data.columns]

# Plot
fig, ax = plt.subplots(figsize=(14, 8))
plot_data.plot(kind='bar', stacked=True, color=plot_colors, ax=ax, width=0.8)

# Styling
ax.set_title('Distribucin de Gnero por Pelcula (Masculino subdividido por Protagonismo)')
ax.set_xlabel('Pelcula')
ax.set_ylabel('Porcentaje (%)')
plt.xticks(rotation=45, ha='right')
plt.legend(title='Categora', bbox_to_anchor=(1.0, 1), loc='upper left')

# 4. Add Labels (Restored Logic: Only show label if > 0)
for container in ax.containers:
    # Create labels: format as percentage if > 0, otherwise leave empty
    labels = [f'{v.get_height():.1f}%' if v.get_height() > 0 else '' for v in container]

    # Apply labels to the center of the bars
    ax.bar_label(container, labels=labels, label_type='center', color='white', fontsize=9, weight='bold')

plt.tight_layout()
plt.savefig('gender_distribution_ghibli.jpg', dpi=300)
plt.show()

# This command will convert the current notebook to an HTML file.
# The output HTML file will be named based on the notebook's filename.
# You can then download it from the Colab file browser.

import os
import glob

print("Debugging: Listing files in /content/")
ls_output = !ls -l /content/
for line in ls_output:
    print(line)

notebook_path = None
# In Colab, the current notebook is usually in /content/ and often the only .ipynb file.
# We'll look for .ipynb files and assume the first one found is the current notebook if only one exists.
notebook_files = glob.glob('/content/*.ipynb')

if len(notebook_files) == 1:
    notebook_path = notebook_files[0]
elif len(notebook_files) > 1:
    print(f"Found multiple .ipynb files: {notebook_files}. Please specify which one to convert.")
    # For now, let's assume the most recently modified one if multiple.
    # This is a heuristic and might not always be correct.
    notebook_path = max(notebook_files, key=os.path.getmtime)
    print(f"Attempting to convert: {notebook_path}")
else:
    print("No .ipynb files found in /content/.")

if notebook_path:
    notebook_name = os.path.basename(notebook_path)
    output_html_name = notebook_name.replace('.ipynb', '.html')
    print(f"Converting '{notebook_name}' to '{output_html_name}'...")
    !jupyter nbconvert --to html "{notebook_path}"
    print(f"Successfully created '{output_html_name}'. You can download it from the file browser.")
else:
    print("Could not determine the current notebook's filename. Please convert manually or specify the filename.")
