import matplotlib.pyplot as plt

# Create a new categorical column that combines Gender and Protagonismo for Masculino
def get_detailed_gender_category(row):
    if row['Genero'] == 'Femenino':
        return 'Femenino'
    elif row['Genero'] == 'Masculino':
        return f"Masculino - {row['Protagonismo']}" # Using ' - ' for readability in legend
    return row['Genero'] # Fallback, though not expected for this dataset

# Apply the function to a copy of the DataFrame to create the new detailed gender column
df_temp_for_plot = df.copy()
df_temp_for_plot['Detailed_Gender_Category'] = df_temp_for_plot.apply(get_detailed_gender_category, axis=1)

# Calculate percentages by 'Pelicula' and 'Detailed_Gender_Category'
detailed_percentages_by_movie = df_temp_for_plot.groupby('Pelicula')['Detailed_Gender_Category'].value_counts(normalize=True) * 100

# Unstack the series to get detailed gender categories as columns
detailed_gender_df_for_plot = detailed_percentages_by_movie.unstack(level='Detailed_Gender_Category').fillna(0)

# Get movie year information for sorting
movie_years = df[['Pelicula', 'Ano']].drop_duplicates().set_index('Pelicula')

# Add 'Ano' to the DataFrame for plotting and sort by it
detailed_gender_df_for_plot = detailed_gender_df_for_plot.merge(movie_years, left_index=True, right_index=True)
detailed_gender_df_for_plot = detailed_gender_df_for_plot.sort_values(by='Ano').drop(columns='Ano')

# Define custom colors
custom_colors = {
    'Femenino': '#F9BDBD',
    'Masculino - Incidental': '#2235AC',
    'Masculino - Protagonista': '#B5BCD6',
    'Masculino - Secundario': '#2EB190'
}

# Create a list of colors for plotting, ensuring order matches DataFrame columns
# Assign a default grey color for any categories not specified by the user
colors_for_plot = [custom_colors.get(col, '#CCCCCC') for col in detailed_gender_df_for_plot.columns]

# Plotting the stacked bar chart
fig, ax = plt.subplots(figsize=(14, 8)) # Use fig, ax for more control and adjust size
plot = detailed_gender_df_for_plot.plot(kind='bar', stacked=True, ax=ax, color=colors_for_plot) # Apply custom colors

# Translate labels and title to Spanish
plt.title('Distribucin de Gnero por Pelcula (Masculino subdividido por Protagonismo)')
plt.xlabel('Pelcula')
plt.ylabel('Porcentaje')
plt.xticks(rotation=45, ha='right') # Rotate x-axis labels for better readability
plt.legend(title='Gnero y Protagonismo', bbox_to_anchor=(1.05, 1), loc='upper left') # Update legend title and position
plt.tight_layout(rect=[0, 0, 0.85, 1]) # Adjust layout to make space for the legend

# Add percentage labels within the bars
for container in ax.containers:
    labels = [f'{w:.1f}%' if (w := v.get_height()) > 0 else '' for v in container]
    ax.bar_label(container, labels=labels, label_type='center', color='white', fontsize=8)

plt.show()
